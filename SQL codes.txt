create database Hotel_MS;
use Hotel_MS;

create table customer(
    Customer_ID int,
    Name varchar(25) NOT NULL,
    Contact varchar(10) NOT NULL,
    Address varchar(20) NOT NULL,
    Primary Key(Customer_ID)
    Constraint num_check check(length(Contact)=10)
);

create table Payment(
    Bill_No int AUTO_INCREMENT,
    Customer_ID int,
    Total_Amount float,
    Primary key(Bill_No),
    Foreign key(Customer_ID) references Customer(Customer_ID)
);

create table Employee(
    Emp_ID varchar(4) NOT NULL,
    Name varchar(25) NOT NULL,
    Contact varchar(10) NOT NULL,
    Address varchar(20) NOT NULL,
    Designation varchar(20) NOT NULL,
    Salary int NOT NULL,
    Primary key(Emp_ID)
    Constraint emp_num_check check(length(Contact)=10);
);

create table Rooms(
    Room_No varchar(4),
    Type varchar(10),
    Price float,
    Status varchar(10),
    Primary key(Room_no)
);

create table Booking(
    Booking_No int AUTO_INCREMENT,
    Check_in date NOT NULL,
    Check_out date NOT NULL,
    Customer_ID int,
    Room_No varchar(4),
    Primary Key(Booking_No),
    Foreign key(Customer_ID) references Customer(Customer_ID),
    Foreign key(Room_No) references Rooms(Room_No)
    Constraint date_check check(Check_out>=Check_in);
);

create table Services(
    Service_No varchar(5),
    Description varchar(10),
    Price float,
    Emp_ID varchar(4),
    Primary key(Service_No),
    Foreign Key(Emp_ID) references Employee(Emp_ID)
);

create table Orders(
    Order_No int AUTO_INCREMENT,
    Customer_ID int, 
    Service_No varchar(5),
    Primary key(Order_No),
    Foreign key(Customer_ID) references Customer(Customer_ID),
    Foreign key(Service_No) references Services(Service_No)
);
